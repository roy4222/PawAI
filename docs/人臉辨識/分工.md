# 人臉辨識模組分工指南

**適用對象**：初階開發者（1 人）  
**前提條件**：系統架構者已完成 MVP（YuNet + IOU Tracker + Go2 wave 鏈路）  
**硬體需求**：❌ 不需要（由架構者提供測試資料）  
**文件版本**：v1.0  
**更新日期**：2026-03-08

---

## 分工原則

因硬體設備（Jetson + RealSense + Go2）集中在架構者手中，初階開發者**僅需筆電即可開發**，透過以下方式隔離硬體依賴：

1. **使用 Mock 物件**模擬硬體行為
2. **使用預錄資料**（rosbag / 影像檔）作為輸入
3. **單元測試驗證**邏輯正確性
4. **架構者整合測試**在真實硬體上驗證

   ```python
   # tests/tracker/test_iou_tracker.py
   def test_iou_no_overlap():

       """完全重疊，IOU 應為 1"""

   def test_tracker_assigns_new_id():
       """新偵測應獲得新的 track_id"""
       tracker = IOUTracker()
       detections = [FaceDetection(bbox=(0, 0, 10, 10), confidence=0.9)]
       tracks = tracker.update(detections, depth_frame=None)
       assert len(tracks) == 1
       assert tracks[0].track_id == 1
   
   def test_tracker_maintains_id():
       """連續幀相同位置，track_id 應保持不變"""
       tracker = IOUTracker()
       # 第一幀
       d1 = [FaceDetection(bbox=(0, 0, 10, 10), confidence=0.9)]
       t1 = tracker.update(d1, None)
       # 第二幀（稍微移動）
       d2 = [FaceDetection(bbox=(2, 2, 12, 12), confidence=0.9)]
       t2 = tracker.update(d2, None)
       assert t1[0].track_id == t2[0].track_id
   ```

### 階段 2：除錯視覺化工具

**任務目標**：開發離線除錯工具，可在沒有硬體的情況下重播和分析人臉追蹤結果

**具體內容**：

1. **離線影像標註器**（3-4 天）
   ```python
   # tools/offline_visualizer.py
   import cv2
   import json
   import numpy as np
   from pathlib import Path
   
   class OfflineFaceVisualizer:
       """
       讀取預錄影像 + 追蹤記錄，產生標註後的除錯影片
       """
       
       def __init__(self):
           self.colors = {}  # track_id -> color
       
       def get_color(self, track_id: int) -> tuple:
           """為每個 track_id 分配固定顏色"""
           if track_id not in self.colors:
               import random
               self.colors[track_id] = (
                   random.randint(100, 255),
                   random.randint(100, 255),
                   random.randint(100, 255)
               )
           return self.colors[track_id]
       
       def draw_frame(self, frame: np.ndarray, tracks: list) -> np.ndarray:
           """在影像上畫出追蹤資訊"""
           for track in tracks:
               x1, y1, x2, y2 = track['bbox']
               color = self.get_color(track['track_id'])
               
               # 畫邊界框
               cv2.rectangle(frame, (x1, y1), (x2, y2), color, 2)
               
               # 畫標籤（ID + 距離）
               label = f"ID:{track['track_id']}"
               if track.get('distance_m'):
                   label += f" {track['distance_m']:.2f}m"
               
               cv2.putText(frame, label, (x1, y1-10),
                          cv2.FONT_HERSHEY_SIMPLEX, 0.6, color, 2)
               
               # 畫中心點
               center = ((x1 + x2) // 2, (y1 + y2) // 2)
               cv2.circle(frame, center, 4, color, -1)
           
           # 畫統計資訊
           stats = f"Tracks: {len(tracks)}"
           cv2.putText(frame, stats, (10, 30),
                      cv2.FONT_HERSHEY_SIMPLEX, 0.7, (255, 255, 255), 2)
           
           return frame
       
       def process_video(self, video_path: str, tracking_log: str, output_path: str):
           """
           主處理函數
           
           Args:
               video_path: 輸入影像路徑（MP4/AVI）
               tracking_log: 追蹤記錄 JSONL 檔案
               output_path: 輸出影像路徑
           """
           # 讀取追蹤記錄
           tracks_by_frame = self._load_tracking_log(tracking_log)
           
           # 處理影像
           cap = cv2.VideoCapture(video_path)
           fps = cap.get(cv2.CAP_PROP_FPS)
           width = int(cap.get(cv2.CAP_PROP_FRAME_WIDTH))
           height = int(cap.get(cv2.CAP_PROP_FRAME_HEIGHT))
           
           fourcc = cv2.VideoWriter_fourcc(*'mp4v')
           out = cv2.VideoWriter(output_path, fourcc, fps, (width, height))
           
           frame_idx = 0
           while True:
               ret, frame = cap.read()
               if not ret:
                   break
               
               # 取得這幀的追蹤資訊
               tracks = tracks_by_frame.get(frame_idx, [])
               
               # 畫上標註
               annotated = self.draw_frame(frame, tracks)
               
               out.write(annotated)
               frame_idx += 1
           
           cap.release()
           out.release()
           print(f"輸出完成: {output_path}")
       
       def _load_tracking_log(self, log_path: str) -> dict:
           """讀取 JSONL 格式的追蹤記錄"""
           tracks_by_frame = {}
           with open(log_path, 'r') as f:
               for line in f:
                   record = json.loads(line)
                   frame_idx = record['frame_idx']
                   tracks_by_frame[frame_idx] = record['tracks']
           return tracks_by_frame
   
   # 使用範例
   if __name__ == "__main__":
       visualizer = OfflineFaceVisualizer()
       visualizer.process_video(
           video_path="test_data/capture.mp4",
           tracking_log="test_data/tracks.jsonl",
           output_path="output/debug_video.mp4"
       )
   ```

2. **追蹤記錄轉換器**（1-2 天）

   ```python
   # tools/bag_to_jsonl.py
   """
   將 rosbag 中的 /state/perception/face 轉換為 JSONL
   供離線視覺化使用
   """
   
   import json
   from rosbag2_py import SequentialReader
   from rclpy.serialization import deserialize_message
   from std_msgs.msg import String
   
   def convert_bag_to_jsonl(bag_path: str, output_path: str):
       """將 rosbag 轉換為 JSONL"""
       reader = SequentialReader()
       reader.open(bag_path)
       
       with open(output_path, 'w') as f:
           while reader.has_next():
               topic, data, timestamp = reader.read_next()
               if topic == '/state/perception/face':
                   msg = deserialize_message(data, String)
                   payload = json.loads(msg.data)
                   f.write(json.dumps(payload, ensure_ascii=False) + '\n')
   ```

### 階段 3：人臉註冊 CLI 工具（前置準備）

**任務目標**：開發離線可用的人臉資料管理工具，為 SFace 身分識別做準備

**具體內容**：

1. **人臉資料管理 CLI**（3-4 天）
   ```python
   # tools/face_database_cli.py
   import argparse
   import json
   import pickle
   from pathlib import Path
   import numpy as np
   
   class FaceDatabaseCLI:
       """人臉資料庫管理命令列工具"""
       
       def __init__(self, db_path: str = "face_database.pkl"):
           self.db_path = Path(db_path)
           self.database = self._load_database()
       
       def _load_database(self) -> dict:
           """載入資料庫，不存在則回傳空資料庫"""
           if self.db_path.exists():
               with open(self.db_path, 'rb') as f:
                   return pickle.load(f)
           return {'persons': {}, 'next_id': 1}
       
       def _save_database(self):
           """儲存資料庫"""
           with open(self.db_path, 'wb') as f:
               pickle.dump(self.database, f)
           print(f"資料庫已儲存: {self.db_path}")
       
       def enroll(self, name: str, embedding_file: str):
           """
           註冊新人物
           
           Args:
               name: 人物姓名
               embedding_file: 128-dim embedding numpy 檔案路徑
           """
           embedding = np.load(embedding_file)
           person_id = f"person_{self.database['next_id']}"
           
           self.database['persons'][person_id] = {
               'name': name,
               'embedding': embedding,
               'created_at': str(datetime.now())
           }
           self.database['next_id'] += 1
           
           self._save_database()
           print(f"已註冊: {name} (ID: {person_id})")
       
       def list_persons(self):
           """列出所有註冊人物"""
           persons = self.database['persons']
           if not persons:
               print("資料庫為空")
               return
           
           print(f"\n{'ID':<15} {'姓名':<20} {'註冊時間'}")
           print("-" * 60)
           for pid, info in persons.items():
               print(f"{pid:<15} {info['name']:<20} {info['created_at']}")
       
       def delete(self, person_id: str):
           """刪除人物"""
           if person_id in self.database['persons']:
               name = self.database['persons'][person_id]['name']
               del self.database['persons'][person_id]
               self._save_database()
               print(f"已刪除: {name} ({person_id})")
           else:
               print(f"找不到: {person_id}")
       
       def export_json(self, output_path: str):
           """匯出為 JSON（不含 embedding，僅供檢視）"""
           export_data = {}
           for pid, info in self.database['persons'].items():
               export_data[pid] = {
                   'name': info['name'],
                   'created_at': info['created_at']
               }
           
           with open(output_path, 'w', encoding='utf-8') as f:
               json.dump(export_data, f, ensure_ascii=False, indent=2)
           print(f"已匯出: {output_path}")
   
   def main():
       parser = argparse.ArgumentParser(description='人臉資料庫管理工具')
       parser.add_argument('--db', default='face_database.pkl', help='資料庫檔案路徑')
       
       subparsers = parser.add_subparsers(dest='command', help='可用指令')
       
       # enroll 指令
       enroll_parser = subparsers.add_parser('enroll', help='註冊新人物')
       enroll_parser.add_argument('name', help='人物姓名')
       enroll_parser.add_argument('embedding', help='embedding numpy 檔案')
       
       # list 指令
       subparsers.add_parser('list', help='列出所有人物')
       
       # delete 指令
       delete_parser = subparsers.add_parser('delete', help='刪除人物')
       delete_parser.add_argument('person_id', help='人物 ID')
       
       # export 指令
       export_parser = subparsers.add_parser('export', help='匯出 JSON')
       export_parser.add_argument('output', help='輸出 JSON 路徑')
       
       args = parser.parse_args()
       
       if not args.command:
           parser.print_help()
           return
       
       cli = FaceDatabaseCLI(args.db)
       
       if args.command == 'enroll':
           cli.enroll(args.name, args.embedding)
       elif args.command == 'list':
           cli.list_persons()
       elif args.command == 'delete':
           cli.delete(args.person_id)
       elif args.command == 'export':
           cli.export_json(args.output)
   
   if __name__ == "__main__":
       from datetime import datetime
       main()
   ```

2. **假 embedding 產生器**（測試用，1 天）
   ```python
   # tools/generate_mock_embeddings.py
   """產生假的 SFace embedding 供測試使用"""
   
   import numpy as np
   from pathlib import Path
   
   def generate_mock_embedding(seed: int = None) -> np.ndarray:
       """產生 128-dim 假 embedding"""
       if seed is not None:
           np.random.seed(seed)
       # SFace embedding 是單位向量
       vec = np.random.randn(128)
       vec = vec / np.linalg.norm(vec)
       return vec
   
   def generate_test_dataset(output_dir: str, num_persons: int = 5):
       """產生測試資料集"""
       output_path = Path(output_dir)
       output_path.mkdir(exist_ok=True)
       
       for i in range(num_persons):
           name = f"person_{i+1}"
           person_dir = output_path / name
           person_dir.mkdir(exist_ok=True)
           
           # 每人 5 個不同 embedding（模擬不同角度）
           for j in range(5):
               emb = generate_mock_embedding(seed=i*100 + j)
               np.save(person_dir / f"embedding_{j}.npy", emb)
       
       print(f"測試資料已產生: {output_path}")
   
   if __name__ == "__main__":
       generate_test_dataset("test_data/mock_embeddings")
   ```
