# 啟動腳本已處理（start_llm_e2e_tmux.sh, start_full_demo_tmux.sh）

# ── 3. rtmlib（RTMPose 備援，必須 --no-deps 避免拉回 CPU onnxruntime）──

aiortc==1.9.0
aiohttp
requests
av
